#include <WiFi.h>
#include <WiFiUdp.h>

const char* ssid = "ASUS";
const char* password = "9137050077";

const char* udpServer = "192.168.1.33";
const int udpPort = 8888;
unsigned char buffer[3731392];
WiFiUDP udp;
int i =0;

void setup() {
  Serial.begin(115200);

  WiFi.begin(ssid, password);
  
  while (WiFi.status() != WL_CONNECTED) {
    delay(1000);
    Serial.println("Connecting to WiFi...");
  }

  Serial.println("Connected to WiFi");
  udp.beginPacket(udpServer, udpPort);
  udp.print("Hello from ESP32");
  Serial.println("Message sended");
  udp.endPacket();

}

void loop() {
  // код для отправки данных или других операций
  if(udp.available()){
    udp.read(buffer, 3731392);
    Serial.println("Packet recieved");
    for(i = 0; i < 20; i++) 
    Serial.print(buffer[i]);
  }
}
